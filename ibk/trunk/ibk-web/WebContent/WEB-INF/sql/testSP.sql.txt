DELIMITER $$

drop procedure if exists data_test;
create procedure data_test()
begin
	IF (NOT EXISTS(SELECT cId FROM tUser WHERE cUser = 'admin' AND md5('admin'))) THEN
		INSERT INTO tUser(cUser, cPassword)
		VALUES('admin', md5('admin'));
	END IF;

end$$

call data_test();
